import * as _ from 'lodash';
import { assertUnreachable } from '../../utils';
import { ThirtyDayForecastAction, ThirtyDayForecastActionType } from './thirty-day-forecast.actions';
import { DefaultThirtyDayForecastState, ThirtyDayForecastState } from './thirty-day-forecast.state';

export const thirtyDayForecastReducer = (
  prevState: ThirtyDayForecastState = DefaultThirtyDayForecastState(),
  action: ThirtyDayForecastAction,
): ThirtyDayForecastState => {
  console.log({ action });
  switch (action.type) {
    case ThirtyDayForecastActionType.GET_THIRTY_DAY_FORECAST_FAILURE:
      return onGetThirtyDayForecastFailure(prevState);
    case ThirtyDayForecastActionType.GET_THIRTY_DAY_FORECAST_REQUEST:
      return onGetThirtyDayForecastRequest(prevState);
    case ThirtyDayForecastActionType.GET_THIRTY_DAY_FORECAST_SUCCESS:
      console.log({ payload: action.payload });

      return onGetThirtyDayForecastSuccess(prevState, action.payload);
    default:
      assertUnreachable(action.type);

      return prevState;
  }
};

const onGetThirtyDayForecastFailure = (prevState: ThirtyDayForecastState) => ({
  ...prevState,
  isLoading: false,
});
 
const onGetThirtyDayForecastRequest = (prevState: ThirtyDayForecastState) => ({
  ...prevState,
  isLoading: true,
});
 
const onGetThirtyDayForecastSuccess = (prevState: ThirtyDayForecastState, newState: ThirtyDayForecastState) => ({
  ...prevState,
  ...newState,
  isLoading: false,
});
