import mocker from 'mocker-data-generator';
import { RootDispatch } from '../../root-action';
import { ThirtyDayForecastState } from './thirty-day-forecast.state';

export enum ThirtyDayForecastActionType {}

export type ThirtyDayForecastAction = {
  type: ThirtyDayForecastActionType;
  payload: ThirtyDayForecastState;
};

export const getThirtyDayForecastRequest = () => ({
  type: ThirtyDayForecastActionType.GET_THIRTY_DAY_FORECAST_REQUEST,
});

export const getThirtyDayForecastSuccess = (
  thirtyDayForecastData: ThirtyDayForecastState['thirtyDayForecastData'],
) => ({
  type: ThirtyDayForecastActionType.GET_THIRTY_DAY_FORECAST_SUCCESS,
  payload: {
    thirtyDayForecastData,
  },
});
 
export const getThirtyDayForecastFailure = () => ({
  type: ThirtyDayForecastActionType.GET_THIRTY_DAY_FORECAST_FAILURE,
});

export const getThirtyDayForecastMockData = async (dispatch: RootDispatch) => {
  dispatch(getThirtyDayForecastRequest());
  const compDaySchema = {
    day: {
      incrementalId: 1,
    },
    comp: {
      faker: 'random.number({"min": -50, "max": 100})',
    },
  };
  try {
    const result = await mocker()
      .schema('lastYearsActuals', compDaySchema, 30)
      .schema('thisYearsActuals', compDaySchema, 21)
      .schema('forecast', compDaySchema, 30)
      .build();
    console.log({ result });
    dispatch(getThirtyDayForecastSuccess(result));
  } catch (e) {
    dispatch(getThirtyDayForecastFailure());
  }
};
