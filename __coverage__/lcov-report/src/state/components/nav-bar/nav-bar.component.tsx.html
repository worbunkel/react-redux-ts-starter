import * as _ from 'lodash';
import * as React from 'react';

type NavLink = {
  linkText: string;
  url: string;
  sublinks?: NavLink[];
};
 
type NavBarProps = {
  logoUrl: string;
  toolName: string;
  links: NavLink[];
};
 
export const NavBar = (props: NavBarProps) => (
  <div>
    {_.map(props.links, link => (
      <div>
        <a href={link.url}>{link.linkText}</a>
        {_.size(link.sublinks) > 0 && <div>{'>'}</div>}
      </div>
    ))}
  </div>
);
