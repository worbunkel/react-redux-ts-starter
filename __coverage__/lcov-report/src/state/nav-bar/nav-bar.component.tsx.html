import * as _ from 'lodash';
import * as React from 'react';
import { connect } from 'react-redux';
import { RootDispatch } from '../root-action';
import { RootState } from '../root-reducer';
import { navigateToLink } from './nav-bar.actions';
import './nav-bar.scss';
import { NavBarState, NavLink } from './nav-bar.state';

const NavBarComponent = (props: NavBarState) => (
  <div>
    {_.map(props.links, link => (
      <div>
        <a href={link.url}>{link.linkText}</a>
        {_.size(link.sublinks) > 0 && <div>{'>'}</div>}
      </div>
    ))}
  </div>
);
 
const mapStateToProps = (state: RootState) => ({
  ...state.navBar,
});

const mapDispatchToProps = (dispatch: RootDispatch) => ({
  navToLink: (navLink: NavLink) => navigateToLink(navLink)(dispatch),
});
 
export const NavBar = connect(
  mapStateToProps,
  mapDispatchToProps,
)(NavBarComponent);
