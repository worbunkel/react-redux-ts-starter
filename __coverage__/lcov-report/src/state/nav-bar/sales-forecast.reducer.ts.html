import * as _ from 'lodash';
import { assertUnreachable } from '../../../utils';
import { NavBarAction, SalesForecastActionType } from './sales-forecast.actions';
import { DefaultNavBarState, ForecastDayDatum, NavBarState } from './sales-forecast.state';

export const navBarReducer = (
  prevState = DefaultSalesForecastState(),
  action: NavBarAction,
): SalesForecastState => {
  switch (action.type) {
    case NavBarActionType.GET_THIRTY_DAY_FORECAST_REQUEST:
      return onGetThirtyDayForecastRequest(prevState);
    case NavBarActionType.GET_THIRTY_DAY_FORECAST_SUCCESS:
      return onGetThirtyDayForecastSuccess(prevState, action.payload.thirtyDayForecastData);
    case NavBarActionType.GET_THIRTY_DAY_FORECAST_FAILURE:
      return onGetThirtyDayForecastFailure(prevState);
    case NavBarActionType.GET_SIXTY_DAY_FORECAST_REQUEST:
      return prevState; // TODO
    case NavBarActionType.GET_SIXTY_DAY_FORECAST_SUCCESS:
      return prevState; // TODO
    case SalesForecastActionType.GET_SIXTY_DAY_FORECAST_FAILURE:
      return prevState; // TODO
    case SalesForecastActionType.GET_NINETY_DAY_FORECAST_REQUEST:
      return prevState; // TODO
    case SalesForecastActionType.GET_NINETY_DAY_FORECAST_SUCCESS:
      return prevState; // TODO
    case SalesForecastActionType.GET_NINETY_DAY_FORECAST_FAILURE:
      return prevState; // TODO
    default:
      assertUnreachable(action.type);

      return prevState;
  }
};

const onGetThirtyDayForecastRequest = (prevState: NavBarState) => ({
  ...prevState,
  isLoading: true,
  users: [],
});
 
const onGetThirtyDayForecastSuccess = (
  prevState: NavBarState,
  thirtyDayForecastData: ForecastDayDatum[] = [],
) => ({
  ...prevState,
  isLoading: false,
  thirtyDayForecastData,
});
 
const onGetThirtyDayForecastFailure = (prevState: NavBarState) => ({
  ...prevState,
  isLoading: false,
});
